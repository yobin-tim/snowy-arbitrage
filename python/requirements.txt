# Dependencies for the Snowy 2.0 arbitrage report.
# Used by the local venv, and by Colab/Binder via `pip install -r requirements.txt`.
# Versions are pinned to the set the committed results were produced with, so a fresh
# install reproduces the report's numbers exactly. Newer versions will most likely work;
# these are simply the known-good ones.
#
# Data:
nemosis==3.8.1     # pulls public AEMO NEMWeb price tables (DISPATCHPRICE)
pandas==3.0.3      # tidy time-series handling
numpy==2.5.1       # arrays / maths
pyarrow==25.0.0    # read/write the cached parquet price snapshots
#
# Optimisation (the perfect-foresight dispatch is a linear program):
cvxpy==1.9.2       # models the LP in readable Python
highspy==1.15.1    # the HiGHS solver cvxpy calls to actually solve it
#
# Plots + notebook tooling:
matplotlib==3.11.0 # all figures
ipykernel==7.3.0   # so the venv shows up as a Jupyter kernel in VS Code
#
# Notebook execution + Binder interface. These run and render the notebook; they do not touch
# the numbers, so their exact versions are not load-bearing for reproducing the results:
nbconvert==7.10.0  # `jupyter nbconvert --execute` runs the notebook end to end (pulls in nbformat)
notebook==7.0.8    # the Jupyter interface Binder launches when someone clicks the badge
